39  /**
40   * Test for {@link Util#isOverridden} method.
41   */
42  public class IsOverriddenTest {
43  
44      @Rule
45      public ErrorCollector errors = new ErrorCollector();
46  
47      /**
48       * Test that a method is found by isOverridden even when it is inherited from an intermediate class.
49       */
50      @Test
51      public void isOverriddenTest() {
52          assertTrue(Util.isOverridden(Base.class, Derived.class, "method"));
53          assertTrue(Util.isOverridden(Base.class, Intermediate.class, "method"));
54          assertFalse(Util.isOverridden(Base.class, Base.class, "method"));
55          assertTrue(Util.isOverridden(Base.class, Intermediate.class, "setX", Object.class));
56          assertTrue(Util.isOverridden(Base.class, Intermediate.class, "getX"));
57      }
58  
59      /**
60       * Negative test.
61       * Trying to check for a method which does not exist in the hierarchy.
62       */
63      @Test
64      public void isOverriddenNegativeTest() {
65          assertThrows(IllegalArgumentException.class, () -> Util.isOverridden(Base.class, Derived.class, "method2"));
66      }
67  
68      /** Specifying a base class that is not a base class should result in an error. */
69      @Test
70      public void badHierarchyIsReported() {
71          assertThrows(IllegalArgumentException.class, () -> Util.isOverridden(Derived.class, Base.class, "method"));
72      }
73  
74      /**
75       * Do not inspect private methods.
76       */
77      @Test
78      public void avoidPrivateMethodsInspection() {
79          assertThrows(IllegalArgumentException.class, () -> Util.isOverridden(Base.class, Intermediate.class, "aPrivateMethod"));
80      }
81  
82      public abstract static class Base<T> {
83          protected abstract void method();
84          private void aPrivateMethod() {}
85          public void setX(T t) {}
86          public T getX() { return null; }
87      }
88      public abstract class Intermediate extends Base<Integer> {
89          protected void method() {}
90          private void aPrivateMethod() {}
91          public void setX(Integer i) {}
92          public Integer getX() { return 0; }
93      }
94      public class Derived extends Intermediate {}

96      @Issue("JENKINS-62723")
97      @Test
98      public void finalOverrides() {
99          errors.checkSucceeds(() -> {assertThat("X1 overrides X.m1", Util.isOverridden(X.class, X1.class, "m1"), is(true)); return null;});
100         errors.checkSucceeds(() -> {assertThat("X1 does not override X.m2", Util.isOverridden(X.class, X1.class, "m2"), is(false)); return null;});
101         errors.checkSucceeds(() -> {assertThat("X2 overrides X.m1", Util.isOverridden(X.class, X2.class, "m1"), is(true)); return null;});
102         errors.checkSucceeds(() -> {assertThat("X2 does not override X.m2", Util.isOverridden(X.class, X2.class, "m2"), is(false)); return null;});
103         errors.checkSucceeds(() -> {assertThat("X3 overrides X.m1", Util.isOverridden(X.class, X3.class, "m1"), is(true)); return null;});
104         errors.checkSucceeds(() -> {assertThat("X3 overrides X.m2", Util.isOverridden(X.class, X3.class, "m2"), is(true)); return null;});
105         errors.checkSucceeds(() -> {assertThat("X4 overrides X.m1", Util.isOverridden(X.class, X4.class, "m1"), is(true)); return null;});
106         errors.checkSucceeds(() -> {assertThat("X4 overrides X.m2", Util.isOverridden(X.class, X4.class, "m2"), is(true)); return null;});
107     }
108     public interface X {
109         void m1();
110         default void m2() {}
111     }
112     public static class X1 implements X {
113         public void m1() {}
114     }
115     public static class X2 implements X {
116         public final void m1() {}
117     }
118     public static class X3 implements X {
119         public void m1() {}
120         @Override
121         public void m2() {}
122     }
123     public static class X4 implements X {
124         public void m1() {}
125         @Override
126         public final void m2() {}
127     }

129     @Issue("JENKINS-62723")
130     @Test
131     public void baseInterface() {
132         // Normal case: classes implementing interface methods
133         errors.checkSucceeds(() -> {assertThat("I1 does not override I1.foo", Util.isOverridden(I1.class, I1.class, "foo"), is(false)); return null;});
134         errors.checkSucceeds(() -> {assertThat("I2 does not override I1.foo", Util.isOverridden(I1.class, I2.class, "foo"), is(false)); return null;});
135         errors.checkSucceeds(() -> {assertThat("C1 does not override I1.foo", Util.isOverridden(I1.class, C1.class, "foo"), is(false)); return null;});
136         errors.checkSucceeds(() -> {assertThat("C2 does not override I1.foo", Util.isOverridden(I1.class, C2.class, "foo"), is(false)); return null;});
137         errors.checkSucceeds(() -> {assertThat("C3 overrides I1.foo", Util.isOverridden(I1.class, C3.class, "foo"), is(true)); return null;});
138         errors.checkSucceeds(() -> {assertThat("C4 overrides I1.foo", Util.isOverridden(I1.class, C4.class, "foo"), is(true)); return null;});
139         // Special case: interfaces providing default implementation of base interface
140         errors.checkSucceeds(() -> {assertThat("I1 does not override I1.bar", Util.isOverridden(I1.class, I1.class, "bar"), is(false)); return null;});
141         errors.checkSucceeds(() -> {assertThat("I2 overrides I1.bar", Util.isOverridden(I1.class, I2.class, "bar"), is(true)); return null;});
142         errors.checkSucceeds(() -> {assertThat("C1 does not override I1.bar", Util.isOverridden(I1.class, C1.class, "bar"), is(false)); return null;});
143         errors.checkSucceeds(() -> {assertThat("C2 overrides I1.bar (via I2)", Util.isOverridden(I1.class, C2.class, "bar"), is(true)); return null;});
144         errors.checkSucceeds(() -> {assertThat("C3 overrides I1.bar (via I2)", Util.isOverridden(I1.class, C3.class, "bar"), is(true)); return null;});
145         errors.checkSucceeds(() -> {assertThat("C4 overrides I1.bar", Util.isOverridden(I1.class, C4.class, "bar"), is(true)); return null;});
146     }
147     private interface I1 {
148         String foo();
149         String bar();
150     }
151     private interface I2 extends I1 {
152         default String bar() { return "default"; }
153     }
154     private abstract static class C1 implements I1 {
155     }
156     private abstract static class C2 extends C1 implements I2 {
157         @Override public abstract String foo();
158     }
159     private abstract static class C3 extends C2 {
160         @Override public String foo() { return "foo"; }
161     }
162     private static class C4 extends C3 implements I1 {
163         @Override public String bar() { return "bar"; }
164     }
