54  /**
55   * Centralizes calls to {@link System#getProperty(String)} and related calls.
56   * This allows us to get values not just from environment variables but also from
57   * the {@link ServletContext}, so properties like {@code hudson.DNSMultiCast.disabled}
58   * can be set in {@code context.xml} and the app server's boot script does not
59   * have to be changed.
60   *
61   * <p>This should be used to obtain hudson/jenkins "app"-level parameters
62   * (e.g. {@code hudson.DNSMultiCast.disabled}), but not for system parameters
63   * (e.g. {@code os.name}).
64   *
65   * <p>If you run multiple instances of Jenkins in the same virtual machine and wish
66   * to obtain properties from {@code context.xml}, make sure these Jenkins instances use
67   * different ClassLoaders. Tomcat, for example, does this automatically. If you do
68   * not use different ClassLoaders, the values of properties specified in
69   * {@code context.xml} is undefined.
70   *
71   * <p>Property access is logged on {@link Level#CONFIG}. Note that some properties
72   * may be accessed by Jenkins before logging is configured properly, so early access to
73   * some properties may not be logged.
74   *
75   * <p>While it looks like it on first glance, this cannot be mapped to {@link EnvVars},
76   * because {@link EnvVars} is only for build variables, not Jenkins itself variables.
77   *
78   * @since 2.236
79   */
80  @SuppressFBWarnings(value = "ST_WRITE_TO_STATIC_FROM_INSTANCE_METHOD", justification = "Currently Jenkins instance may have one ond only one context")
81  public class SystemProperties {
82  
83      @FunctionalInterface
84      private interface Handler {
85          @CheckForNull
86          String getString(String key);
87      }
88  
89      private static final Handler NULL_HANDLER = key -> null;
90  
91      private static @NonNull Handler handler = NULL_HANDLER;
92  
93      // declared in WEB-INF/web.xml
94      @Restricted(NoExternalUse.class)
95      public static final class Listener implements ServletContextListener, OnMaster {
96  
97          /**
98           * Called by the servlet container to initialize the {@link ServletContext}.
99           */
100         @Override
101         public void contextInitialized(ServletContextEvent event) {
102             ServletContext theContext = event.getServletContext();
103             handler = key -> {
104                 if (StringUtils.isNotBlank(key)) {
105                     try {
106                         return theContext.getInitParameter(key);
107                     } catch (SecurityException ex) {
108                         // Log exception and go on
109                         LOGGER.log(Level.CONFIG, "Access to the property {0} is prohibited", key);
110                     }
111                 }
112                 return null;
113             };
114         }
115 
116         @Override
117         public void contextDestroyed(ServletContextEvent event) {
118             handler = NULL_HANDLER;
119         }
120 
121     }

123     private static final Set<String> ALLOW_ON_AGENT = Collections.synchronizedSet(new HashSet<>());
124 
125     /**
126      * Mark a key whose value should be made accessible in agent JVMs.
127      *
128      * @param key Property key to be explicitly allowed
129      */
130     public static void allowOnAgent(String key) {
131         ALLOW_ON_AGENT.add(key);
132     }

134     @Extension
135     @Restricted(NoExternalUse.class)
136     public static final class AgentCopier extends ComputerListener {
137         @Override
138         public void preOnline(Computer c, Channel channel, FilePath root, TaskListener listener) throws IOException, InterruptedException {
139             channel.call(new CopySystemProperties());
140         }
141         private static final class CopySystemProperties extends MasterToSlaveCallable<Void, RuntimeException> {
142             private static final long serialVersionUID = 1;
143             private final Map<String, String> snapshot;
144             CopySystemProperties() {
145                 // Take a snapshot of those system properties and context variables available on the master at the time the agent starts which have been whitelisted for that purpose.
146                 snapshot = new HashMap<>();
147                 for (String key : ALLOW_ON_AGENT) {
148                     snapshot.put(key, getString(key));
149                 }
150                 LOGGER.log(Level.FINE, "taking snapshot of {0}", snapshot);
151             }
152             @Override
153             public Void call() throws RuntimeException {
154                 handler = new CopiedHandler(snapshot);
155                 return null;
156             }
157         }
158         private static final class CopiedHandler implements Handler {
159             private final Map<String, String> snapshot;
160             CopiedHandler(Map<String, String> snapshot) {
161                 this.snapshot = snapshot;
162             }
163             @Override
164             public String getString(String key) {
165                 return snapshot.get(key);
166             }
167         }
168     }

177     /**
178      * Gets the system property indicated by the specified key.
179      * This behaves just like {@link System#getProperty(java.lang.String)}, except that it
180      * also consults the {@link ServletContext}'s "init" parameters.
181      * 
182      * @param      key   the name of the system property.
183      * @return     the string value of the system property,
184      *             or {@code null} if there is no property with that key.
185      *
186      * @exception  NullPointerException if {@code key} is {@code null}.
187      * @exception  IllegalArgumentException if {@code key} is empty.
188      */
189     @CheckForNull
190     public static String getString(String key) {
191         return getString(key, null);
192     }
193 
194     /**
195      * Gets the system property indicated by the specified key, or a default value.
196      * This behaves just like {@link System#getProperty(java.lang.String, java.lang.String)}, except
197      * that it also consults the {@link ServletContext}'s "init" parameters.
198      * 
199      * @param      key   the name of the system property.
200      * @param      def   a default value.
201      * @return     the string value of the system property,
202      *             or {@code null} if the property is missing and the default value is {@code null}.
203      *
204      * @exception  NullPointerException if {@code key} is {@code null}.
205      * @exception  IllegalArgumentException if {@code key} is empty.
206      */
207     public static String getString(String key, @CheckForNull String def) {
208         return getString(key, def, Level.CONFIG);
209     }

211     /**
212      * Gets the system property indicated by the specified key, or a default value.
213      * This behaves just like {@link System#getProperty(java.lang.String, java.lang.String)}, except
214      * that it also consults the {@link ServletContext}'s "init" parameters.
215      *
216      * @param      key   the name of the system property.
217      * @param      def   a default value.
218      * @param      logLevel the level of the log if the provided key is not found.
219      * @return     the string value of the system property,
220      *             or {@code null} if the property is missing and the default value is {@code null}.
221      *
222      * @exception  NullPointerException if {@code key} is {@code null}.
223      * @exception  IllegalArgumentException if {@code key} is empty.
224      */
225     public static String getString(String key, @CheckForNull String def, Level logLevel) {
226         String value = System.getProperty(key); // keep passing on any exceptions
227         if (value != null) {
228             if (LOGGER.isLoggable(logLevel)) {
229                 LOGGER.log(logLevel, "Property (system): {0} => {1}", new Object[] {key, value});
230             }
231             return value;
232         } 
233         
234         value = handler.getString(key);
235         if (value != null) {
236             if (LOGGER.isLoggable(logLevel)) {
237                 LOGGER.log(logLevel, "Property (context): {0} => {1}", new Object[]{key, value});
238             }
239             return value;
240         }
241         
242         value = def;
243         if (LOGGER.isLoggable(logLevel)) {
244             LOGGER.log(logLevel, "Property (default): {0} => {1}", new Object[] {key, value});
245         }
246         return value;
247     }

249     /**
250       * Returns {@code true} if the system property
251       * named by the argument exists and is equal to the string
252       * {@code "true"}. If the system property does not exist, return
253       * {@code "false"}. if a property by this name exists in the {@link ServletContext}
254       * and is equal to the string {@code "true"}.
255       * 
256       * This behaves just like {@link Boolean#getBoolean(java.lang.String)}, except that it
257       * also consults the {@link ServletContext}'s "init" parameters.
258       * 
259       * @param   name   the system property name.
260       * @return  the {@code boolean} value of the system property.
261       */  
262     public static boolean getBoolean(String name) {
263         return getBoolean(name, false);
264     }
265 
266     /**
267       * Returns {@code true} if the system property
268       * named by the argument exists and is equal to the string
269       * {@code "true"}, or a default value. If the system property does not exist, return
270       * {@code "true"} if a property by this name exists in the {@link ServletContext}
271       * and is equal to the string {@code "true"}. If that property does not
272       * exist either, return the default value.
273       * 
274       * This behaves just like {@link Boolean#getBoolean(java.lang.String)} with a default
275       * value, except that it also consults the {@link ServletContext}'s "init" parameters.
276       * 
277       * @param   name   the system property name.
278       * @param   def   a default value.
279       * @return  the {@code boolean} value of the system property.
280       */
281     public static boolean getBoolean(String name, boolean def) {
282         String v = getString(name);
283        
284         if (v != null) {
285             return Boolean.parseBoolean(v);
286         }
287         return def;
288     }
289 
290     /**
291      * Returns {@link Boolean#TRUE} if the named system property exists and is equal to the string {@code "true}
292      * (ignoring case), returns {@link Boolean#FALSE} if the system property exists and doesn't equal {@code "true}
293      * otherwise returns {@code null} if the named system property does not exist.
294      *
295      * @param name the system property name.
296      * @return {@link Boolean#TRUE}, {@link Boolean#FALSE} or {@code null}
297      * @since 2.16
298      */
299     @CheckForNull
300     public static Boolean optBoolean(String name) {
301         String v = getString(name);
302         return v == null ? null : Boolean.parseBoolean(v);
303     }

321      * Determines the integer value of the system property with the
322      * specified name, or a default value.
323      *
324      * This behaves just like {@code Integer.getInteger(String,Integer)}, except that it
325      * also consults the {@code ServletContext}'s "init" parameters. If neither exist,
326      * return the default value.
327      *
328      * @param   name property name.
329      * @param   def   a default value.
330      * @return  the {@code Integer} value of the property.

332      *          Result may be {@code null} only if the default value is {@code null}.

334     public static Integer getInteger(String name, Integer def) {
335         return getInteger(name, def, Level.CONFIG);

338     /**
339       * Determines the integer value of the system property with the
340       * specified name, or a default value.
341       * 
342       * This behaves just like {@code Integer.getInteger(String,Integer)}, except that it
343       * also consults the {@code ServletContext}'s "init" parameters. If neither exist,
344       * return the default value. 
345       * 
346       * @param   name property name.
347       * @param   def   a default value.
348       * @param   logLevel the level of the log if the provided system property name cannot be decoded into Integer.
349       * @return  the {@code Integer} value of the property.
350       *          If the property is missing, return the default value.
351       *          Result may be {@code null} only if the default value is {@code null}.
352       */
353     public static Integer getInteger(String name, Integer def, Level logLevel) {
354         String v = getString(name);
355        
356         if (v != null) {
357             try {
358                 return Integer.decode(v);
359             } catch (NumberFormatException e) {
360                 // Ignore, fallback to default
361                 if (LOGGER.isLoggable(logLevel)) {
362                     LOGGER.log(logLevel, "Property. Value is not integer: {0} => {1}", new Object[] {name, v});
363                 }
364             }
365         }
366         return def;
367     }
