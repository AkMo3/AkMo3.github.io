86  	public static ChangeCorrectionProposal getInvertVariableProposal(CodeActionParams params, IInvocationContext context, ASTNode covering, boolean returnAsCommand) {
87  		// cursor should be placed on variable name
88  		if (!(covering instanceof SimpleName)) {
89  			return null;
90  		}
91  		SimpleName coveringName = (SimpleName) covering;
92  		if (!coveringName.isDeclaration()) {
93  			return null;
94  		}
95  		// prepare bindings
96  		final IBinding variableBinding = coveringName.resolveBinding();
97  		if (!(variableBinding instanceof IVariableBinding)) {
98  			return null;
99  		}
100 		IVariableBinding binding = (IVariableBinding) variableBinding;
101 		if (binding.isField()) {
102 			return null;
103 		}
104 		// we operate only on boolean variable
105 		if (!isBoolean(coveringName)) {
106 			return null;
107 		}
108 
109 		String label = CorrectionMessages.AdvancedQuickAssistProcessor_inverseBooleanVariable;
110 		if (returnAsCommand) {
111 			return new CUCorrectionCommandProposal(label, CodeActionKind.Refactor, context.getCompilationUnit(), IProposalRelevance.INVERSE_BOOLEAN_VARIABLE, RefactorProposalUtility.APPLY_REFACTORING_COMMAND_ID,
112 					Arrays.asList(INVERT_VARIABLE_COMMAND, params));
113 		}
114 
115 		final AST ast = covering.getAST();
116 		// find linked nodes
117 		final MethodDeclaration method = ASTResolving.findParentMethodDeclaration(covering);
118 		SimpleName[] linkedNodes = LinkedNodeFinder.findByBinding(method, variableBinding);
119 		//
120 		final ASTRewrite rewrite = ASTRewrite.create(ast);
121 		// create proposal
122 		final String KEY_NAME = "name"; //$NON-NLS-1$
123 		final LinkedCorrectionProposal proposal = new LinkedCorrectionProposal(label, CodeActionKind.Refactor, context.getCompilationUnit(), rewrite, IProposalRelevance.INVERSE_BOOLEAN_VARIABLE);
124 		// prepare new variable identifier
125 		final String oldIdentifier = coveringName.getIdentifier();
126 		final String notString = Messages.format(CorrectionMessages.AdvancedQuickAssistProcessor_negatedVariableName, ""); //$NON-NLS-1$
127 		final String newIdentifier;
128 		if (oldIdentifier.startsWith(notString)) {
129 			int notLength = notString.length();
130 			if (oldIdentifier.length() > notLength) {
131 				newIdentifier = Character.toLowerCase(oldIdentifier.charAt(notLength)) + oldIdentifier.substring(notLength + 1);
132 			} else {
133 				newIdentifier = oldIdentifier;
134 			}
135 		} else {
136 			newIdentifier = Messages.format(CorrectionMessages.AdvancedQuickAssistProcessor_negatedVariableName, Character.toUpperCase(oldIdentifier.charAt(0)) + oldIdentifier.substring(1));
137 		}
138 		//
139 		proposal.addLinkedPositionProposal(KEY_NAME, newIdentifier);
140 		proposal.addLinkedPositionProposal(KEY_NAME, oldIdentifier);
141 		// iterate over linked nodes and replace variable references with negated reference
142 		final HashSet<SimpleName> renamedNames = new HashSet<>();
143 		for (int i = 0; i < linkedNodes.length; i++) {
144 			SimpleName name = linkedNodes[i];
145 			if (renamedNames.contains(name)) {
146 				continue;
147 			}
148 			// prepare new name with new identifier
149 			SimpleName newName = ast.newSimpleName(newIdentifier);
150 			proposal.addLinkedPosition(rewrite.track(newName), name == coveringName, KEY_NAME);
151 			//
152 			StructuralPropertyDescriptor location = name.getLocationInParent();
153 			if (location == SingleVariableDeclaration.NAME_PROPERTY) {
154 				// set new name
155 				rewrite.replace(name, newName, null);
156 			} else if (location == Assignment.LEFT_HAND_SIDE_PROPERTY) {
157 				Assignment assignment = (Assignment) name.getParent();
158 				Expression expression = assignment.getRightHandSide();
159 				int exStart = expression.getStartPosition();
160 				int exEnd = exStart + expression.getLength();
161 				// collect all names that are used in assignments
162 				HashSet<SimpleName> overlapNames = new HashSet<>();
163 				for (int j = 0; j < linkedNodes.length; j++) {
164 					SimpleName name2 = linkedNodes[j];
165 					if (name2 == null) {
166 						continue;
167 					}
168 					int name2Start = name2.getStartPosition();
169 					if (exStart <= name2Start && name2Start < exEnd) {
170 						overlapNames.add(name2);
171 					}
172 				}
173 				// prepare inverted expression
174 				SimpleNameRenameProvider provider = new SimpleNameRenameProvider() {
175 					@Override
176 					public SimpleName getRenamed(SimpleName simpleName) {
177 						if (simpleName.resolveBinding() == variableBinding) {
178 							renamedNames.add(simpleName);
179 							return ast.newSimpleName(newIdentifier);
180 						}
181 						return null;
182 					}
183 				};
184 				Expression inversedExpression = getInversedExpression(rewrite, expression, provider);
185 				// if any name was not renamed during expression inverting, we can not already rename it, so fail to create assist
186 				for (Iterator<SimpleName> iter = overlapNames.iterator(); iter.hasNext();) {
187 					Object o = iter.next();
188 					if (!renamedNames.contains(o)) {
189 						return null;
190 					}
191 				}
192 				// check operator and replace if needed
193 				Assignment.Operator operator = assignment.getOperator();
194 				if (operator == Assignment.Operator.BIT_AND_ASSIGN) {
195 					Assignment newAssignment = ast.newAssignment();
196 					newAssignment.setLeftHandSide(newName);
197 					newAssignment.setRightHandSide(inversedExpression);
198 					newAssignment.setOperator(Assignment.Operator.BIT_OR_ASSIGN);
199 					rewrite.replace(assignment, newAssignment, null);
200 				} else if (operator == Assignment.Operator.BIT_OR_ASSIGN) {
201 					Assignment newAssignment = ast.newAssignment();
202 					newAssignment.setLeftHandSide(newName);
203 					newAssignment.setRightHandSide(inversedExpression);
204 					newAssignment.setOperator(Assignment.Operator.BIT_AND_ASSIGN);
205 					rewrite.replace(assignment, newAssignment, null);
206 				} else {
207 					rewrite.replace(expression, inversedExpression, null);
208 					// set new name
209 					rewrite.replace(name, newName, null);
210 				}
211 			} else if (location == VariableDeclarationFragment.NAME_PROPERTY) {
212 				// replace initializer for variable
213 				VariableDeclarationFragment vdf = (VariableDeclarationFragment) name.getParent();
214 				Expression expression = vdf.getInitializer();
215 				if (expression != null) {
216 					rewrite.replace(expression, getInversedExpression(rewrite, expression), null);
217 				}
218 				// set new name
219 				rewrite.replace(name, newName, null);
220 			} else if (name.getParent() instanceof PrefixExpression && ((PrefixExpression) name.getParent()).getOperator() == PrefixExpression.Operator.NOT) {
221 				rewrite.replace(name.getParent(), newName, null);
222 			} else {
223 				PrefixExpression expression = ast.newPrefixExpression();
224 				expression.setOperator(PrefixExpression.Operator.NOT);
225 				expression.setOperand(newName);
226 				rewrite.replace(name, expression, null);
227 			}
228 		}
229 
230 		return proposal;
231 	}

239 	private static boolean getInverseConditionProposals(CodeActionParams params, IInvocationContext context, ASTNode covering, ArrayList<ASTNode> coveredNodes, Collection<ChangeCorrectionProposal> proposals) {
240 		if (proposals == null) {
241 			return false;
242 		}
243 
244 		final AST ast;
245 		final ASTRewrite rewrite;
246 
247 		if (context.getSelectionLength() == 0) {
248 			Expression foundExpression = null;
249 			while (covering instanceof Expression) {
250 				Expression booleanExpression = getBooleanExpression(covering);
251 				if (booleanExpression != null) {
252 					foundExpression = getBooleanExpression(covering);
253 				}
254 				covering = covering.getParent();
255 			}
256 			if (foundExpression == null) {
257 				return false;
258 			}
259 
260 			ast = foundExpression.getAST();
261 			rewrite = ASTRewrite.create(ast);
262 
263 			Expression inversedExpression = getInversedExpression(rewrite, foundExpression);
264 			rewrite.replace(foundExpression, inversedExpression, null);
265 		} else {
266 			if (coveredNodes.isEmpty()) {
267 				return false;
268 			}
269 			ast = covering.getAST();
270 			rewrite = ASTRewrite.create(ast);
271 			// check sub-expressions in fully covered nodes
272 			boolean hasChanges = false;
273 			for (Iterator<ASTNode> iter = coveredNodes.iterator(); iter.hasNext();) {
274 				ASTNode covered = iter.next();
275 				Expression coveredExpression = getBooleanExpression(covered);
276 				if (coveredExpression != null) {
277 					Expression inversedExpression = getInversedExpression(rewrite, coveredExpression);
278 					rewrite.replace(coveredExpression, inversedExpression, null);
279 					hasChanges = true;
280 				}
281 			}
282 
283 			if (!hasChanges) {
284 				return false;
285 			}
286 		}
287 
288 		// add correction proposal
289 		String label = CorrectionMessages.AdvancedQuickAssistProcessor_inverseConditions_description;
290 		ASTRewriteCorrectionProposal proposal = new ASTRewriteCorrectionProposal(label, CodeActionKind.Refactor, context.getCompilationUnit(), rewrite, IProposalRelevance.INVERSE_CONDITIONS);
291 		proposals.add(proposal);
292 		return true;
293 	}

296 		PrefixExpression prefixExpression = ast.newPrefixExpression();
297 		prefixExpression.setOperator(PrefixExpression.Operator.NOT);
298 		ParenthesizedExpression parenthesizedExpression = getParenthesizedExpression(ast, (Expression) rewrite.createCopyTarget(expression));
299 		prefixExpression.setOperand(parenthesizedExpression);

303 	private static boolean isBoolean(Expression expression) {
304 		ITypeBinding typeBinding = expression.resolveTypeBinding();
305 		AST ast = expression.getAST();
306 		return typeBinding == ast.resolveWellKnownType("boolean") //$NON-NLS-1$
307 				|| typeBinding == ast.resolveWellKnownType("java.lang.Boolean"); //$NON-NLS-1$
308 	}

310 	private static Expression getBooleanExpression(ASTNode node) {
311 		if (!(node instanceof Expression)) {
312 			return null;
313 		}
314 
315 		// check if the node is a location where it can be negated
316 		StructuralPropertyDescriptor locationInParent = node.getLocationInParent();
317 		if (locationInParent == QualifiedName.NAME_PROPERTY) {
318 			node = node.getParent();
319 			locationInParent = node.getLocationInParent();
320 		}
321 		while (locationInParent == ParenthesizedExpression.EXPRESSION_PROPERTY) {
322 			node = node.getParent();
323 			locationInParent = node.getLocationInParent();
324 		}
325 		Expression expression = (Expression) node;
326 		if (!isBoolean(expression)) {
327 			return null;
328 		}
329 		if (expression.getParent() instanceof InfixExpression) {
330 			return expression;
331 		}
332 		if (locationInParent == Assignment.RIGHT_HAND_SIDE_PROPERTY || locationInParent == IfStatement.EXPRESSION_PROPERTY || locationInParent == WhileStatement.EXPRESSION_PROPERTY || locationInParent == DoStatement.EXPRESSION_PROPERTY
333 				|| locationInParent == ReturnStatement.EXPRESSION_PROPERTY || locationInParent == ForStatement.EXPRESSION_PROPERTY || locationInParent == AssertStatement.EXPRESSION_PROPERTY
334 				|| locationInParent == MethodInvocation.ARGUMENTS_PROPERTY || locationInParent == ConstructorInvocation.ARGUMENTS_PROPERTY || locationInParent == SuperMethodInvocation.ARGUMENTS_PROPERTY
335 				|| locationInParent == EnumConstantDeclaration.ARGUMENTS_PROPERTY || locationInParent == SuperConstructorInvocation.ARGUMENTS_PROPERTY || locationInParent == ClassInstanceCreation.ARGUMENTS_PROPERTY
336 				|| locationInParent == ConditionalExpression.EXPRESSION_PROPERTY || locationInParent == PrefixExpression.OPERAND_PROPERTY) {
337 			return expression;
338 		}
339 		return null;
340 	}

343 		InfixExpression newExpression = rewrite.getAST().newInfixExpression();
344 		newExpression.setOperator(newOperator);
345 		newExpression.setLeftOperand(getRenamedNameCopy(provider, rewrite, expression.getLeftOperand()));
346 		newExpression.setRightOperand(getRenamedNameCopy(provider, rewrite, expression.getRightOperand()));
347 		return newExpression;
348 	}

350 	private static Expression parenthesizeIfRequired(Expression operand, int newOperatorPrecedence) {
351 		if (newOperatorPrecedence > OperatorPrecedence.getExpressionPrecedence(operand)) {
352 			return getParenthesizedExpression(operand.getAST(), operand);
353 		}
354 		return operand;
355 	}

363 	private static Expression getInversedAndOrExpression(ASTRewrite rewrite, InfixExpression infixExpression, Operator newOperator, SimpleNameRenameProvider provider) {
364 		InfixExpression newExpression = rewrite.getAST().newInfixExpression();
365 		newExpression.setOperator(newOperator);
366 
367 		int newOperatorPrecedence = OperatorPrecedence.getOperatorPrecedence(newOperator);
368 
369 		Expression leftOperand = getInversedExpression(rewrite, infixExpression.getLeftOperand(), provider);
370 		newExpression.setLeftOperand(parenthesizeIfRequired(leftOperand, newOperatorPrecedence));
371 
372 		Expression rightOperand = getInversedExpression(rewrite, infixExpression.getRightOperand(), provider);
373 		newExpression.setRightOperand(parenthesizeIfRequired(rightOperand, newOperatorPrecedence));
374 
375 		List<Expression> extraOperands = infixExpression.extendedOperands();
376 		List<Expression> newExtraOperands = newExpression.extendedOperands();
377 		for (int i = 0; i < extraOperands.size(); i++) {
378 			Expression extraOperand = getInversedExpression(rewrite, extraOperands.get(i), provider);
379 			newExtraOperands.add(parenthesizeIfRequired(extraOperand, newOperatorPrecedence));
380 		}
381 		return newExpression;
382 	}

388 	private static Expression getRenamedNameCopy(SimpleNameRenameProvider provider, ASTRewrite rewrite, Expression expression) {
389 		if (provider != null) {
390 			if (expression instanceof SimpleName) {
391 				SimpleName name = (SimpleName) expression;
392 				SimpleName newName = provider.getRenamed(name);
393 				if (newName != null) {
394 					return newName;
395 				}
396 			}
397 		}
398 		return (Expression) rewrite.createCopyTarget(expression);
399 	}

406 	private static Expression getInversedExpression(ASTRewrite rewrite, Expression expression, SimpleNameRenameProvider provider) {
407 		AST ast = rewrite.getAST();
408 
409 		if (expression instanceof BooleanLiteral) {
410 			return ast.newBooleanLiteral(!((BooleanLiteral) expression).booleanValue());
411 		}
412 		if (expression instanceof InfixExpression) {
413 			InfixExpression infixExpression = (InfixExpression) expression;
414 			InfixExpression.Operator operator = infixExpression.getOperator();
415 			if (operator == InfixExpression.Operator.LESS) {
416 				return getInversedInfixExpression(rewrite, infixExpression, InfixExpression.Operator.GREATER_EQUALS, provider);
417 			}
418 			if (operator == InfixExpression.Operator.GREATER) {
419 				return getInversedInfixExpression(rewrite, infixExpression, InfixExpression.Operator.LESS_EQUALS, provider);
420 			}
421 			if (operator == InfixExpression.Operator.LESS_EQUALS) {
422 				return getInversedInfixExpression(rewrite, infixExpression, InfixExpression.Operator.GREATER, provider);
423 			}
424 			if (operator == InfixExpression.Operator.GREATER_EQUALS) {
425 				return getInversedInfixExpression(rewrite, infixExpression, InfixExpression.Operator.LESS, provider);
426 			}
427 			if (operator == InfixExpression.Operator.EQUALS) {
428 				return getInversedInfixExpression(rewrite, infixExpression, InfixExpression.Operator.NOT_EQUALS, provider);
429 			}
430 			if (operator == InfixExpression.Operator.NOT_EQUALS) {
431 				return getInversedInfixExpression(rewrite, infixExpression, InfixExpression.Operator.EQUALS, provider);
432 			}
433 			if (operator == InfixExpression.Operator.CONDITIONAL_AND) {
434 				return getInversedAndOrExpression(rewrite, infixExpression, InfixExpression.Operator.CONDITIONAL_OR, provider);
435 			}
436 			if (operator == InfixExpression.Operator.CONDITIONAL_OR) {
437 				return getInversedAndOrExpression(rewrite, infixExpression, InfixExpression.Operator.CONDITIONAL_AND, provider);
438 			}
439 			if (operator == InfixExpression.Operator.AND) {
440 				return getInversedAndOrExpression(rewrite, infixExpression, InfixExpression.Operator.OR, provider);
441 			}
442 			if (operator == InfixExpression.Operator.OR) {
443 				return getInversedAndOrExpression(rewrite, infixExpression, InfixExpression.Operator.AND, provider);
444 			}
445 			if (operator == InfixExpression.Operator.XOR) {
446 				return getInversedNotExpression(rewrite, expression, ast);
447 			}
448 		}
449 		if (expression instanceof PrefixExpression) {
450 			PrefixExpression prefixExpression = (PrefixExpression) expression;
451 			if (prefixExpression.getOperator() == PrefixExpression.Operator.NOT) {
452 				Expression operand = prefixExpression.getOperand();
453 				if ((operand instanceof ParenthesizedExpression) && NecessaryParenthesesChecker.canRemoveParentheses(operand, expression.getParent(), expression.getLocationInParent())) {
454 					operand = ((ParenthesizedExpression) operand).getExpression();
455 				}
456 				Expression renamedNameCopy = getRenamedNameCopy(provider, rewrite, operand);
457 				if (renamedNameCopy instanceof InfixExpression) {
458 					InfixExpression infixExpression = (InfixExpression) renamedNameCopy;
459 					infixExpression.setOperator(((InfixExpression) operand).getOperator());
460 				}
461 				return renamedNameCopy;
462 			}
463 		}
464 		if (expression instanceof InstanceofExpression) {
465 			return getInversedNotExpression(rewrite, expression, ast);
466 		}
467 		if (expression instanceof ParenthesizedExpression) {
468 			ParenthesizedExpression parenthesizedExpression = (ParenthesizedExpression) expression;
469 			Expression innerExpression = parenthesizedExpression.getExpression();
470 			while (innerExpression instanceof ParenthesizedExpression) {
471 				innerExpression = ((ParenthesizedExpression) innerExpression).getExpression();
472 			}
473 			if (innerExpression instanceof InstanceofExpression) {
474 				return getInversedExpression(rewrite, innerExpression, provider);
475 			}
476 			parenthesizedExpression = getParenthesizedExpression(ast, getInversedExpression(rewrite, innerExpression, provider));
477 			return parenthesizedExpression;
478 		}
479 		if (expression instanceof ConditionalExpression) {
480 			ConditionalExpression conditionalExpression = (ConditionalExpression) expression;
481 			ConditionalExpression newExpression = ast.newConditionalExpression();
482 			newExpression.setExpression((Expression) rewrite.createCopyTarget(conditionalExpression.getExpression()));
483 			newExpression.setThenExpression(getInversedExpression(rewrite, conditionalExpression.getThenExpression()));
484 			newExpression.setElseExpression(getInversedExpression(rewrite, conditionalExpression.getElseExpression()));
485 			return newExpression;
486 		}
487 
488 		PrefixExpression prefixExpression = ast.newPrefixExpression();
489 		prefixExpression.setOperator(PrefixExpression.Operator.NOT);
490 		Expression renamedNameCopy = getRenamedNameCopy(provider, rewrite, expression);
491 		if (NecessaryParenthesesChecker.needsParentheses(renamedNameCopy, prefixExpression, PrefixExpression.OPERAND_PROPERTY)) {
492 			renamedNameCopy = getParenthesizedExpression(ast, renamedNameCopy);
493 		}
494 		prefixExpression.setOperand(renamedNameCopy);
495 		return prefixExpression;
496 	}

498 	public static boolean getSplitAndConditionProposals(IInvocationContext context, ASTNode node, Collection<ChangeCorrectionProposal> resultingCollections) {
499 		Operator andOperator = InfixExpression.Operator.CONDITIONAL_AND;
500 		// check that user invokes quick assist on infix expression
501 		if (!(node instanceof InfixExpression)) {
502 			return false;
503 		}
504 		InfixExpression infixExpression = (InfixExpression) node;
505 		if (infixExpression.getOperator() != andOperator) {
506 			return false;
507 		}
508 		int offset = isOperatorSelected(infixExpression, context.getSelectionOffset(), context.getSelectionLength());
509 		if (offset == -1) {
510 			return false;
511 		}
512 
513 		// check that infix expression belongs to IfStatement
514 		Statement statement = ASTResolving.findParentStatement(node);
515 		if (!(statement instanceof IfStatement)) {
516 			return false;
517 		}
518 		IfStatement ifStatement = (IfStatement) statement;
519 
520 		// check that infix expression is part of first level && condition of IfStatement
521 		InfixExpression topInfixExpression = infixExpression;
522 		while (topInfixExpression.getParent() instanceof InfixExpression && ((InfixExpression) topInfixExpression.getParent()).getOperator() == andOperator) {
523 			topInfixExpression = (InfixExpression) topInfixExpression.getParent();
524 		}
525 		if (ifStatement.getExpression() != topInfixExpression) {
526 			return false;
527 		}
528 		//
529 		if (resultingCollections == null) {
530 			return true;
531 		}
532 		AST ast = ifStatement.getAST();
533 		ASTRewrite rewrite = ASTRewrite.create(ast);
534 
535 		// prepare left and right conditions
536 		Expression[] newOperands = { null, null };
537 		breakInfixOperationAtOperation(rewrite, topInfixExpression, andOperator, offset, true, newOperands);
538 
539 		Expression leftCondition = newOperands[0];
540 		Expression rightCondition = newOperands[1];
541 
542 		// replace conditions in outer IfStatement
543 		rewrite.set(ifStatement, IfStatement.EXPRESSION_PROPERTY, leftCondition, null);
544 
545 		// prepare inner IfStatement
546 		IfStatement innerIf = ast.newIfStatement();
547 
548 		innerIf.setExpression(rightCondition);
549 		innerIf.setThenStatement((Statement) rewrite.createMoveTarget(ifStatement.getThenStatement()));
550 		Block innerBlock = ast.newBlock();
551 		innerBlock.statements().add(innerIf);
552 
553 		Statement elseStatement = ifStatement.getElseStatement();
554 		if (elseStatement != null) {
555 			innerIf.setElseStatement((Statement) rewrite.createCopyTarget(elseStatement));
556 		}
557 
558 		// replace outer thenStatement
559 		rewrite.replace(ifStatement.getThenStatement(), innerBlock, null);
560 
561 		// add correction proposal
562 		String label = CorrectionMessages.AdvancedQuickAssistProcessor_splitAndCondition_description;
563 		ASTRewriteCorrectionProposal proposal = new ASTRewriteCorrectionProposal(label, CodeActionKind.QuickFix, context.getCompilationUnit(), rewrite, IProposalRelevance.SPLIT_AND_CONDITION);
564 		resultingCollections.add(proposal);
565 		return true;
566 	}

599 			return true;
600 		}

603

605 		Expression[] newOperands = { null, null };
606 		breakInfixOperationAtOperation(rewrite, topInfixExpression, orOperator, offset, true, newOperands);
607 
608 		Expression leftCondition = newOperands[0];
609 		Expression rightCondition = newOperands[1];
610 
611 		// prepare first statement
612 		rewrite.replace(ifStatement.getExpression(), leftCondition, null);
613 
614 		IfStatement secondIf = ast.newIfStatement();
615 		secondIf.setExpression(rightCondition);
616 		secondIf.setThenStatement((Statement) rewrite.createCopyTarget(ifStatement.getThenStatement()));
617

619 		if (elseStatement == null) {
620 			rewrite.set(ifStatement, IfStatement.ELSE_STATEMENT_PROPERTY, secondIf, null);
621 		} else {

623 			secondIf.setElseStatement((Statement) rewrite.createMoveTarget(elseStatement));
624 		}

627 		String label = CorrectionMessages.AdvancedQuickAssistProcessor_splitOrCondition_description;

629 		resultingCollections.add(proposal);

633 	private static boolean isSelectingOperator(ASTNode n1, ASTNode n2, int offset, int length) {
634 		// between the nodes
635 		if (offset + length <= n2.getStartPosition() && offset >= ASTNodes.getExclusiveEnd(n1)) {
636 			return true;
637 		}
638 		// or exactly select the node (but not with infix expressions)
639 		if (n1.getStartPosition() == offset && ASTNodes.getExclusiveEnd(n2) == offset + length) {
640 			if (n1 instanceof InfixExpression || n2 instanceof InfixExpression) {
641 				return false;
642 			}
643 			return true;
644 		}
645 		return false;
646 	}

648 	private static int isOperatorSelected(InfixExpression infixExpression, int offset, int length) {
649 		ASTNode left = infixExpression.getLeftOperand();
650 		ASTNode right = infixExpression.getRightOperand();
651 
652 		if (isSelectingOperator(left, right, offset, length)) {
653 			return ASTNodes.getExclusiveEnd(left);
654 		}
655 		List<Expression> extended = infixExpression.extendedOperands();
656 		for (int i = 0; i < extended.size(); i++) {
657 			left = right;
658 			right = extended.get(i);
659 			if (isSelectingOperator(left, right, offset, length)) {
660 				return ASTNodes.getExclusiveEnd(left);
661 			}
662 		}
663 		return -1;
664 	}

666 	/*
667 	 * Breaks an infix operation with possible extended operators at the given operator and returns the new left and right operands.
668 	 * a & b & c   ->  [[a' & b' ] & c' ]   (c' == copy of c)
669 	 */
670 	private static void breakInfixOperationAtOperation(ASTRewrite rewrite, Expression expression, Operator operator, int operatorOffset, boolean removeParentheses, Expression[] res) {
671 		if (expression.getStartPosition() + expression.getLength() <= operatorOffset) {
672 			// add to the left
673 			res[0] = combineOperands(rewrite, res[0], expression, removeParentheses, operator);
674 			return;
675 		}
676 		if (operatorOffset <= expression.getStartPosition()) {
677 			// add to the right
678 			res[1] = combineOperands(rewrite, res[1], expression, removeParentheses, operator);
679 			return;
680 		}
681 		if (!(expression instanceof InfixExpression)) {
682 			throw new IllegalArgumentException("Cannot break up non-infix expression"); //$NON-NLS-1$
683 		}
684 		InfixExpression infixExpression = (InfixExpression) expression;
685 		if (infixExpression.getOperator() != operator) {
686 			throw new IllegalArgumentException("Incompatible operator"); //$NON-NLS-1$
687 		}
688 		breakInfixOperationAtOperation(rewrite, infixExpression.getLeftOperand(), operator, operatorOffset, removeParentheses, res);
689 		breakInfixOperationAtOperation(rewrite, infixExpression.getRightOperand(), operator, operatorOffset, removeParentheses, res);
690 
691 		List<Expression> extended = infixExpression.extendedOperands();
692 		for (int i = 0; i < extended.size(); i++) {
693 			breakInfixOperationAtOperation(rewrite, extended.get(i), operator, operatorOffset, removeParentheses, res);
694 		}
695 	}

697 	private static Expression combineOperands(ASTRewrite rewrite, Expression existing, Expression originalNode, boolean removeParentheses, Operator operator) {
698 		if (existing == null && removeParentheses) {
699 			while (originalNode instanceof ParenthesizedExpression) {
700 				originalNode = ((ParenthesizedExpression) originalNode).getExpression();
701 			}
702 		}
703 		Expression newRight = (Expression) rewrite.createMoveTarget(originalNode);
704 		if (originalNode instanceof InfixExpression) {
705 			((InfixExpression) newRight).setOperator(((InfixExpression) originalNode).getOperator());
706 		}
707 
708 		if (existing == null) {
709 			return newRight;
710 		}
711 		AST ast = rewrite.getAST();
712 		InfixExpression infix = ast.newInfixExpression();
713 		infix.setOperator(operator);
714 		infix.setLeftOperand(existing);
715 		infix.setRightOperand(newRight);
716 		return infix;
717 	}
