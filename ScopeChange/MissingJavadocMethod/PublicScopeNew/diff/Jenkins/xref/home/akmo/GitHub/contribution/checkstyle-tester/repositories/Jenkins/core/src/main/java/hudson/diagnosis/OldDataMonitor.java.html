71  /**
72   * Tracks whether any data structure changes were corrected when loading XML,
73   * that could be resaved to migrate that data to the new format.
74   *
75   * @author Alan.Harder@Sun.Com
76   */
77  @Extension @Symbol("oldData")
78  public class OldDataMonitor extends AdministrativeMonitor {
79      private static final Logger LOGGER = Logger.getLogger(OldDataMonitor.class.getName());

81      private ConcurrentMap<SaveableReference,VersionRange> data = new ConcurrentHashMap<>();

83      /**
84       * Gets instance of the monitor.
85       * @param j Jenkins instance
86       * @return Monitor instance

88       *              It should never happen since the monitor is located in the core.

108     public Map<Saveable,VersionRange> getData() {
109         Map<Saveable,VersionRange> r = new HashMap<>();
110         for (Map.Entry<SaveableReference,VersionRange> entry : this.data.entrySet()) {
111             Saveable s = entry.getKey().get();
112             if (s != null) {
113                 r.put(s, entry.getValue());
114             }
115         }
116         return r;
117     }

119     private static void remove(Saveable obj, boolean isDelete) {
120         Jenkins j = Jenkins.get();
121         OldDataMonitor odm = get(j);
122         try (ACLContext ctx = ACL.as2(ACL.SYSTEM2)) {
123             odm.data.remove(referTo(obj));
124             if (isDelete && obj instanceof Job<?, ?>) {
125                 for (Run r : ((Job<?, ?>) obj).getBuilds()) {
126                     odm.data.remove(referTo(r));
127                 }
128             }
129         }
130     }

158     /**
159      * Inform monitor that some data in a deprecated format has been loaded,
160      * and converted in-memory to a new structure.
161      * @param obj Saveable object; calling save() on this object will persist
162      *            the data in its new format to disk.
163      * @param version Hudson release when the data structure changed.
164      */
165     public static void report(Saveable obj, String version) {
166         OldDataMonitor odm = get(Jenkins.get());
167         try {
168             SaveableReference ref = referTo(obj);
169             while (true) {
170                 VersionRange vr = odm.data.get(ref);
171                 if (vr != null && odm.data.replace(ref, vr, new VersionRange(vr, version, null))) {
172                     break;
173                 } else if (odm.data.putIfAbsent(ref, new VersionRange(null, version, null)) == null) {
174                     break;
175                 }
176             }
177         } catch (IllegalArgumentException ex) {
178             LOGGER.log(Level.WARNING, "Bad parameter given to OldDataMonitor", ex);
179         }
180     }

182     /**
183      * Inform monitor that some data in a deprecated format has been loaded, during
184      * XStream unmarshalling when the Saveable containing this object is not available.
185      * @param context XStream unmarshalling context
186      * @param version Hudson release when the data structure changed.
187      */
188     public static void report(UnmarshallingContext context, String version) {
189         RobustReflectionConverter.addErrorInContext(context, new ReportException(version));
190     }

192     private static class ReportException extends Exception {
193         private String version;
194         private ReportException(String version) {
195             this.version = version;
196         }
197     }

199     /**
200      * Inform monitor that some unreadable data was found while loading.
201      * @param obj Saveable object; calling save() on this object will discard the unreadable data.
202      * @param errors Exception(s) thrown while loading, regarding the unreadable classes/fields.
203      */
204     public static void report(Saveable obj, Collection<Throwable> errors) {
205         StringBuilder buf = new StringBuilder();
206         int i = 0;
207         for (Throwable e : errors) {
208             if (e instanceof ReportException) {
209                 report(obj, ((ReportException)e).version);
210             } else {
211                 if (Main.isUnitTest) {
212                     LOGGER.log(Level.INFO, "Trouble loading " + obj, e);
213                 }
214                 if (++i > 1) buf.append(", ");
215                 buf.append(e.getClass().getSimpleName()).append(": ").append(e.getMessage());
216             }
217         }
218         if (buf.length() == 0) return;
219         Jenkins j = Jenkins.getInstanceOrNull();
220         if (j == null) { // Need this path, at least for unit tests, but also in case of very broken startup
221             // Startup failed, something is very broken, so report what we can.
222             for (Throwable t : errors) {
223                 LOGGER.log(Level.WARNING, "could not read " + obj + " (and Jenkins did not start up)", t);
224             }
225             return;
226         }
227         OldDataMonitor odm = get(j);
228         SaveableReference ref = referTo(obj);
229         while (true) {
230             VersionRange vr = odm.data.get(ref);
231             if (vr != null && odm.data.replace(ref, vr, new VersionRange(vr, null, buf.toString()))) {
232                 break;
233             } else if (odm.data.putIfAbsent(ref, new VersionRange(null, null, buf.toString())) == null) {
234                 break;
235             }
236         }
237     }

239     public static class VersionRange {
240         private static VersionNumber currentVersion = Jenkins.getVersion();
241 
242         final VersionNumber min;
243         final VersionNumber max;
244         final boolean single;
245         public final String extra;
246 
247         public VersionRange(VersionRange previous, String version, String extra) {
248             if (previous == null) {
249                 min = max = version != null ? new VersionNumber(version) : null;
250                 this.single = true;
251                 this.extra = extra;
252             } else if (version == null) {
253                 min = previous.min;
254                 max = previous.max;
255                 single = previous.single;
256                 this.extra = extra;
257             } else {
258                 VersionNumber ver = new VersionNumber(version);
259                 if (previous.min == null || ver.isOlderThan(previous.min)) {
260                     this.min = ver;
261                 } else {
262                     this.min = previous.min;
263                 }
264                 if (previous.max == null || ver.isNewerThan(previous.max)) {
265                     this.max = ver;
266                 } else {
267                     this.max = previous.max;
268                 }
269                 this.single = this.max.isNewerThan(this.min);
270                 this.extra = extra;
271             }
272         }
273 
274         @Override
275         public String toString() {
276             return min==null ? "" : min.toString() + (single ? "" : " - " + max.toString());
277         }
278 
279         /**
280          * Does this version range contain a version more than the given number of releases ago?
281          * @param threshold Number of releases
282          * @return True if the major version# differs or the minor# differs by â‰¥ threshold
283          */
284         public boolean isOld(int threshold) {
285             return currentVersion != null && min != null && (currentVersion.getDigitAt(0) > min.getDigitAt(0)
286                     || (currentVersion.getDigitAt(0) == min.getDigitAt(0)
287                     && currentVersion.getDigitAt(1) - min.getDigitAt(1) >= threshold));
288         }
289 
290     }

292     /**
293      * Sorted list of unique max-versions in the data set.  For select list in jelly.
294      */
295     @Restricted(NoExternalUse.class)
296     public Iterator<VersionNumber> getVersionList() {
297         TreeSet<VersionNumber> set = new TreeSet<>();
298         for (VersionRange vr : data.values()) {
299             if (vr.max != null) {
300                 set.add(vr.max);
301             }
302         }
303         return set.iterator();
304     }

319     /**
320      * Save all or some of the files to persist data in the new forms.
321      * Remove those items from the data map.
322      */
323     @RequirePOST
324     public HttpResponse doUpgrade(StaplerRequest req, StaplerResponse rsp) {
325         final String thruVerParam = req.getParameter("thruVer");
326         final VersionNumber thruVer = thruVerParam.equals("all") ? null : new VersionNumber(thruVerParam);
327 
328         saveAndRemoveEntries(entry -> {
329             VersionNumber version = entry.getValue().max;
330             return version != null && (thruVer == null || !version.isNewerThan(thruVer));
331         });
332 
333         return HttpResponses.forwardToPreviousPage();
334     }
335 
336     /**
337      * Save all files containing only unreadable data (no data upgrades), which discards this data.
338      * Remove those items from the data map.
339      */
340     @RequirePOST
341     public HttpResponse doDiscard(StaplerRequest req, StaplerResponse rsp) {
342         saveAndRemoveEntries(entry -> entry.getValue().max == null);
343 
344         return HttpResponses.forwardToPreviousPage();
345     }

347     private void saveAndRemoveEntries(Predicate<Map.Entry<SaveableReference, VersionRange>> matchingPredicate) {
348         /*
349          * Note that there a race condition here: we acquire the lock and get localCopy which includes some
350          * project (say); then we go through our loop and save that project; then someone POSTs a new
351          * config.xml for the project with some old data, causing remove to be called and the project to be
352          * added to data (in the new version); then we hit the end of this method and the project is removed
353          * from data again, even though it again has old data.
354          *
355          * In practice this condition is extremely unlikely, and not a major problem even if it
356          * does occur: just means the user will be prompted to discard less than they should have been (and
357          * would see the warning again after next restart).
358          */
359         List<SaveableReference> removed = new ArrayList<>();
360         for (Map.Entry<SaveableReference,VersionRange> entry : data.entrySet()) {
361             if (matchingPredicate.test(entry)) {
362                 Saveable s = entry.getKey().get();
363                 if (s != null) {
364                     try {
365                         s.save();
366                     } catch (Exception x) {
367                         LOGGER.log(Level.WARNING, "failed to save " + s, x);
368                     }
369                 }
370                 removed.add(entry.getKey());
371             }
372         }
373 
374         data.keySet().removeAll(removed);
375     }

381     /** Reference to a saveable object that need not actually hold it in heap. */
382     private interface SaveableReference {
383         @CheckForNull Saveable get();
384         // must also define equals, hashCode
385     }
386 
387     private static SaveableReference referTo(Saveable s) {
388         if (s instanceof Run) {
389             Job parent = ((Run) s).getParent();
390             if (Jenkins.get().getItemByFullName(parent.getFullName()) == parent) {
391                 return new RunSaveableReference((Run) s);
392             }
393         }
394         return new SimpleSaveableReference(s);
395     }
396 
397     private static final class SimpleSaveableReference implements SaveableReference {
398         private final Saveable instance;
399         SimpleSaveableReference(Saveable instance) {
400             this.instance = instance;
401         }
402         @Override public Saveable get() {
403             return instance;
404         }
405         @Override public int hashCode() {
406             return instance.hashCode();
407         }
408         @Override public boolean equals(Object obj) {
409             return obj instanceof SimpleSaveableReference && instance.equals(((SimpleSaveableReference) obj).instance);
410         }
411     }
412 
413     // could easily make an ItemSaveableReference, but Jenkins holds all these strongly, so why bother
414 
415     private static final class RunSaveableReference implements SaveableReference {
416         private final String id;
417         RunSaveableReference(Run<?,?> r) {
418             id = r.getExternalizableId();
419         }
420         @Override public Saveable get() {
421             try {
422                 return Run.fromExternalizableId(id);
423             } catch (IllegalArgumentException x) {
424                 // Typically meaning the job or build was since deleted.
425                 LOGGER.log(Level.FINE, null, x);
426                 return null;
427             }
428         }
429         @Override public int hashCode() {
430             return id.hashCode();
431         }
432         @Override public boolean equals(Object obj) {
433             return obj instanceof RunSaveableReference && id.equals(((RunSaveableReference) obj).id);
434         }
435     }
