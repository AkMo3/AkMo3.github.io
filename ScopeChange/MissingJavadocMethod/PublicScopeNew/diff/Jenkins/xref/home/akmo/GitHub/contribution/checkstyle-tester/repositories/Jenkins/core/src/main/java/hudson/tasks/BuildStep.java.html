55  /**
56   * One step of the whole build process.
57   *
58   * <h2>Persistence</h2>
59   * <p>
60   * These objects are persisted as a part of {@link Project} by XStream.
61   * The save operation happens without any notice, and the restore operation
62   * happens without calling the constructor, just like Java serialization.
63   *
64   * <p>
65   * So generally speaking, derived classes should use instance variables
66   * only for keeping configuration. You can still store objects you use
67   * for processing, like a parser of some sort, but they need to be marked
68   * as {@code transient}, and the code needs to be aware that they might
69   * be null (which is the case when you access the field for the first time
70   * the object is restored.)
71   *
72   * <h2>Lifecycle</h2>
73   * <p>
74   * Build steps are instantiated when the user saves the job configuration, and sticks
75   * around in memory until the job configuration is overwritten.
76   *
77   * @author Kohsuke Kawaguchi
78   */
79  public interface BuildStep {
80  
81      /**
82       * Runs before the build begins.
83       *
84       * @return
85       *      true if the build can continue, false if there was an error
86       *      and the build needs to be aborted.
87       *      <p>
88       *      Using the return value to indicate success/failure should
89       *      be considered deprecated, and implementations are encouraged
90       *      to throw {@link AbortException} to indicate a failure.
91       */
92      boolean prebuild( AbstractBuild<?,?> build, BuildListener listener );
93  
94      /**
95       * Runs the step over the given build and reports the progress to the listener.
96       *
97       * <p>
98       * A plugin can contribute the action object to {@link Build#getActions()}
99       * so that a 'report' becomes a part of the persisted data of {@link Build}.
100      * This is how JUnit plugin attaches the test report to a build page, for example.
101      *
102      * <p>When this build step needs to make (direct or indirect) permission checks to {@link ACL}
103      * (for example, to locate other projects by name, build them, or access their artifacts)
104      * then it must be run under a specific {@link Authentication}.
105      * In such a case, the implementation should check whether {@link Jenkins#getAuthentication2} is {@link ACL#SYSTEM2},
106      * and if so, replace it for the duration of this step with {@link Jenkins#ANONYMOUS}.
107      * (Either using {@link ACL#impersonate2}, or by making explicit calls to {@link ACL#hasPermission2(Authentication, Permission)}.)
108      * This would typically happen when no {@link QueueItemAuthenticator} was available, configured, and active.
109      *
110      * @return
111      *      true if the build can continue, false if there was an error
112      *      and the build needs to be aborted.
113      *      <p>
114      *      Using the return value to indicate success/failure should
115      *      be considered deprecated, and implementations are encouraged
116      *      to throw {@link AbortException} to indicate a failure.
117      *
118      * @throws InterruptedException
119      *      If the build is interrupted by the user (in an attempt to abort the build.)
120      *      Normally the {@link BuildStep} implementations may simply forward the exception
121      *      it got from its lower-level functions.
122      * @throws IOException
123      *      If the implementation wants to abort the processing when an {@link IOException}
124      *      happens, it can simply propagate the exception to the caller. This will cause
125      *      the build to fail, with the default error message.
126      *      Implementations are encouraged to catch {@link IOException} on its own to
127      *      provide a better error message, if it can do so, so that users have better
128      *      understanding on why it failed.
129      */
130     boolean perform(AbstractBuild<?,?> build, Launcher launcher, BuildListener listener) throws InterruptedException, IOException;
131 
132     /**
133      * @deprecated as of 1.341.
134      *      Use {@link #getProjectActions(AbstractProject)} instead.
135      */
136     @Deprecated
137     Action getProjectAction(AbstractProject<?,?> project);

139     /**
140      * Returns action objects if this {@link BuildStep} has actions
141      * to contribute to a {@link Project}.
142      *
143      * <p>
144      * {@link Project} calls this method for every {@link BuildStep} that
145      * it owns when the rendering is requested.
146      *
147      * <p>
148      * This action can have optional {@code jobMain.jelly} view, which will be
149      * aggregated into the main panel of the job top page. The jelly file
150      * should have an {@code <h2>} tag that shows the section title, followed by some
151      * block elements to render the details of the section.
152      *
153      * @param project
154      *      {@link Project} that owns this build step,
155      *      since {@link BuildStep} object doesn't usually have this "parent" pointer.
156      *
157      * @return
158      *      can be empty but never null.
159      */
160     @NonNull
161     Collection<? extends Action> getProjectActions(AbstractProject<?,?> project);
162 
163 
164     /**
165      * Declares the scope of the synchronization monitor this {@link BuildStep} expects from outside.
166      *
167      * <p>
168      * This method is introduced for preserving compatibility with plugins written for earlier versions of Hudson,
169      * which never run multiple builds of the same job in parallel. Such plugins often assume that the outcome
170      * of the previous build is completely available, which is no longer true when we do concurrent builds.
171      *
172      * <p>
173      * To minimize the necessary code change for such plugins, {@link BuildStep} implementations can request
174      * Hudson to externally perform synchronization before executing them. This behavior is as follows:
175      *
176      * <dl>
177      * <dt>{@link BuildStepMonitor#BUILD}
178      * <dd>
179      * This {@link BuildStep} is only executed after the previous build is fully
180      * completed (thus fully restoring the earlier semantics of one build at a time.)
181      *
182      * <dt>{@link BuildStepMonitor#STEP}
183      * <dd>
184      * This {@link BuildStep} is only executed after the same step in the previous build is completed.
185      * For build steps that use a weaker assumption and only rely on the output from the same build step of
186      * the early builds, this improves the concurrency.
187      *
188      * <dt>{@link BuildStepMonitor#NONE}
189      * <dd>
190      * No external synchronization is performed on this build step. This is the most efficient, and thus
191      * <b>the recommended value for newer plugins</b>. Wherever necessary, you can directly use {@link CheckPoint}s
192      * to perform necessary synchronizations.
193      * </dl>
194      *
195      * <h2>Migrating Older Implementation</h2>
196      * <p>
197      * If you are migrating {@link BuildStep} implementations written for earlier versions of Hudson,
198      * here's what you can do:
199      *
200      * <ul>
201      * <li>
202      * To demand the backward compatible behavior from Jenkins, leave this method unoverridden,
203      * and make no other changes to the code. This will prevent users from reaping the benefits of concurrent
204      * builds, but at least your plugin will work correctly, and therefore this is a good easy first step.
205      * <li>
206      * If your build step doesn't use anything from a previous build (for example, if you don't even call
207      * {@link Run#getPreviousBuild()}), then you can return {@link BuildStepMonitor#NONE} without making further
208      * code changes and you are done with migration.
209      * <li>
210      * If your build step only depends on {@link Action}s that you added in the previous build by yourself,
211      * then you only need {@link BuildStepMonitor#STEP} scope synchronization. Return it from this method
212      * ,and you are done with migration without any further code changes.
213      * <li>
214      * If your build step makes more complex assumptions, return {@link BuildStepMonitor#NONE} and use
215      * {@link CheckPoint}s directly in your code. The general idea is to call {@link CheckPoint#block()} before
216      * you try to access the state from the previous build.
217      * </ul>
218      *
219      * @since 1.319
220      */
221     default BuildStepMonitor getRequiredMonitorService() {
222         return BuildStepMonitor.BUILD;
223     }

226      * List of all installed builders.
227      *
228      * Builders are invoked to perform the build itself.

237     /**
238      * List of all installed publishers.
239      *
240      * Publishers are invoked after the build is completed, normally to perform
241      * some post-actions on build results, such as sending notifications, collecting
242      * results, etc.
243      *
244      * @see PublisherList#addNotifier(Descriptor)
245      * @see PublisherList#addRecorder(Descriptor)
246      *
247      * @deprecated as of 1.286.
248      *      Use {@link Publisher#all()} for read access, and use
249      *      {@link Extension} for registration.
250      */
251     @Deprecated
252     PublisherList PUBLISHERS = new PublisherList();
253 
254     /**
255      * List of publisher descriptor.
256      */
257     final class PublisherList extends AbstractList<Descriptor<Publisher>> {
258         /**
259          * {@link Descriptor}s are actually stored in here.
260          * Since {@link PublisherList} lives longer than {@link jenkins.model.Jenkins} we cannot directly use {@link ExtensionList}.
261          */
262         private final DescriptorList<Publisher> core = new DescriptorList<>(Publisher.class);
263 
264         /**
265          * For descriptors that are manually registered, remember what kind it was since
266          * older plugins don't extend from neither {@link Recorder} nor {@link Notifier}.
267          */
268         /*package*/ static final WeakHashMap<Descriptor<Publisher>,Class<? extends Publisher>/*either Recorder.class or Notifier.class*/>
269                 KIND = new WeakHashMap<>();
270 
271         private PublisherList() {
272         }
273 
274         /**
275          * Adds a new publisher descriptor, which (generally speaking)
276          * shouldn't alter the build result, but just report the build result
277          * by some means, such as e-mail, IRC, etc.
278          *
279          * <p>
280          * This method adds the descriptor after all the "recorders".
281          *
282          * @see #addRecorder(Descriptor)
283          */
284         public void addNotifier( Descriptor<Publisher> d ) {
285             KIND.put(d,Notifier.class);
286             core.add(d);
287         }
288         
289         /**
290          * Adds a new publisher descriptor, which (generally speaking)
291          * alter the build result based on some artifacts of the build.
292          *
293          * <p>
294          * This method adds the descriptor before all the "notifiers".
295          *
296          * @see #addNotifier(Descriptor) 
297          */
298         public void addRecorder( Descriptor<Publisher> d ) {
299             KIND.put(d,Recorder.class);
300             core.add(d);
301         }
302 
303         @Override
304         public boolean add(Descriptor<Publisher> d) {
305             return !contains(d) && core.add(d);
306         }
307 
308         @Override
309         public void add(int index, Descriptor<Publisher> d) {
310             if(!contains(d)) core.add(d);
311         }
312 
313         public Descriptor<Publisher> get(int index) {
314             return core.get(index);
315         }
316 
317         public int size() {
318             return core.size();
319         }
320 
321         @Override
322         public Iterator<Descriptor<Publisher>> iterator() {
323             return core.iterator();
324         }
325 
326         @Override
327         public boolean remove(Object o) {
328             return core.remove(o);
329         }
330     }
