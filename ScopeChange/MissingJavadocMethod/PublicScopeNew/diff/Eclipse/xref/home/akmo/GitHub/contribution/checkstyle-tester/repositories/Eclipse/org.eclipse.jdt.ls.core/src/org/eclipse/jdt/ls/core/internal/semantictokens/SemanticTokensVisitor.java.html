53  	public SemanticTokensVisitor(CompilationUnit cu) {

59  	private class SemanticToken {
60  		private final TokenType tokenType;
61  		private final int tokenModifiers;
62  		private final int offset;
63  		private final int length;

65  		public SemanticToken(int offset, int length, TokenType tokenType, int tokenModifiers) {
66  			this.offset = offset;
67  			this.length = length;
68  			this.tokenType = tokenType;
69  			this.tokenModifiers = tokenModifiers;
70  		}
71  
72  		public TokenType getTokenType() {
73  			return tokenType;
74  		}
75  
76  		public int getTokenModifiers() {
77  			return tokenModifiers;
78  		}
79  
80  		public int getOffset() {
81  			return offset;
82  		}
83  
84  		public int getLength() {
85  			return length;
86  		}

89  	public SemanticTokens getSemanticTokens() {
90  		return new SemanticTokens(encodedTokens());
91  	}

93  	private int[] encodedTokens() {
94  		int numTokens = tokens.size();
95  		int[] data = new int[numTokens * 5];
96  		int currentLine = 0;
97  		int currentColumn = 0;
98  		for (int i = 0; i < numTokens; i++) {
99  			SemanticToken token = tokens.get(i);
100 			int line = cu.getLineNumber(token.getOffset()) - 1;
101 			int column = cu.getColumnNumber(token.getOffset());
102 			int deltaLine = line - currentLine;
103 			if (deltaLine != 0) {
104 				currentLine = line;
105 				currentColumn = 0;
106 			}
107 			int deltaColumn = column - currentColumn;
108 			currentColumn = column;
109 			// Disallow duplicate/conflict token (if exists)
110 			if (deltaLine != 0 || deltaColumn != 0) {
111 				int tokenTypeIndex = token.getTokenType().ordinal();
112 				int tokenModifiers = token.getTokenModifiers();
113 
114 				int offset = i * 5;
115 				data[offset] = deltaLine;
116 				data[offset + 1] = deltaColumn;
117 				data[offset + 2] = token.getLength();
118 				data[offset + 3] = tokenTypeIndex;
119 				data[offset + 4] = tokenModifiers;
120 			}
121 		}
122 		return data;
123 	}

125 	/**
126 	 * "Static" modifiers which are always added by {@link #addToken(int, int, TokenType, int)}.
127 	 * Modifiers can be set or removed at any time during the visitation process, and as such
128 	 * will be applied for all nodes visited during the time a static modifier is set.
129 	 *
130 	 * Setting and removing the modifiers is done via bitwise OR/AND, see
131 	 * {@link TokenModifier#bitmask} and {@link TokenModifier#inverseBitmask}
132 	 */
133 	private int staticModifiers = 0;
134 
135 	/**
136 	 * Adds a semantic token to the list of tokens being collected by this
137 	 * semantic token visitor.
138 	 *
139 	 * @param offset The document offset of the semantic token.
140 	 * @param length The length of the semantic token.
141 	 * @param tokenType The type of the semantic token.
142 	 * @param modifiers The bitwise OR of the semantic token modifiers, see {@link TokenModifier#bitmask}.
143 	 *
144 	 * @apiNote This method is order-dependent because of {@link #encodedTokens()}.
145 	 * If semantic tokens are not added in the order they appear in the document,
146 	 * the encoding algorithm might discard them.
147 	 */
148 	private void addToken(int offset, int length, TokenType tokenType, int modifiers) {
149 		tokens.add(new SemanticToken(offset, length, tokenType, modifiers | staticModifiers));
150 	}

152 	/**
153 	 * Adds a semantic token to the list of tokens being collected by this
154 	 * semantic token visitor.
155 	 *
156 	 * @param node The AST node representing the location of the semantic token.
157 	 * @param tokenType The type of the semantic token.
158 	 * @param modifiers The bitwise OR of the semantic token modifiers, see {@link TokenModifier#bitmask}.
159 	 *
160 	 * @apiNote This method is order-dependent because of {@link #encodedTokens()}.
161 	 * If semantic tokens are not added in the order they appear in the document,
162 	 * the encoding algorithm might discard them.
163 	 */
164 	private void addToken(ASTNode node, TokenType tokenType, int modifiers) {
165 		addToken(node.getStartPosition(), node.getLength(), tokenType, modifiers);
166 	}

168 	/**
169 	 * Adds a semantic token to the list of tokens being collected by this
170 	 * semantic token visitor.
171 	 *
172 	 * @param node The AST node representing the location of the semantic token.
173 	 * @param tokenType The type of the semantic token.
174 	 *
175 	 * @apiNote This method is order-dependent because of {@link #encodedTokens()}.
176 	 * If semantic tokens are not added in the order they appear in the document,
177 	 * the encoding algorithm might discard them.
178 	 */
179 	private void addToken(ASTNode node, TokenType tokenType) {
180 		addToken(node, tokenType, 0);
181 	}

183 	@Override
184 	public boolean visit(TypeLiteral node) {
185 		acceptNode(node.getType());
186 		// The last 5 characters of a TypeLiteral are the "class" keyword
187 		int offset = node.getStartPosition() + node.getLength() - 5;
188 		addToken(offset, 5, TokenType.KEYWORD, 0);
189 		return false;
190 	}

192 	@Override
193 	public boolean visit(Javadoc node) {
194 		staticModifiers |= TokenModifier.DOCUMENTATION.bitmask;
195 		return super.visit(node);
196 	}
197 
198 	@Override
199 	public void endVisit(Javadoc node) {
200 		staticModifiers &= TokenModifier.DOCUMENTATION.inverseBitmask;
201 		super.endVisit(node);
202 	}
203 
204 	@Override
205 	public boolean visit(TagElement node) {
206 		if (node.getTagName() != null) {
207 			// If the token is nested, we need to skip the { character
208 			int offset = node.isNested() ? node.getStartPosition() + 1 : node.getStartPosition();
209 			addToken(offset, node.getTagName().length(), TokenType.KEYWORD, 0);
210 		}
211 		acceptNodeList(node.fragments());
212 		return false;
213 	}
214 
215 	@Override
216 	public boolean visit(PackageDeclaration node) {
217 		acceptNode(node.getJavadoc());
218 		acceptNodeList(node.annotations());
219 		addTokenToSimpleNamesOfName(node.getName(), TokenType.NAMESPACE);
220 		return false;
221 	}
222 
223 	@Override
224 	public boolean visit(ImportDeclaration node) {
225 		staticModifiers |= TokenModifier.IMPORT_DECLARATION.bitmask;
226 
227 		IBinding binding = node.resolveBinding();
228 		if (binding == null || binding instanceof IPackageBinding) {
229 			addTokenToSimpleNamesOfName(node.getName(), TokenType.NAMESPACE);
230 		}
231 		else {
232 			nonPackageNameOfImportDeclarationVisitor(node.getName());
233 		}
234 
235 		staticModifiers &= TokenModifier.IMPORT_DECLARATION.inverseBitmask;
236 		return false;
237 	}

239 	/**
240 	 * {@link NodeVisitor} for {@link Name} nodes inside an import declaration that are not package names.
241 	 *
242 	 * @param nonPackageName
243 	 */
244 	private void nonPackageNameOfImportDeclarationVisitor(Name nonPackageName) {
245 		if (nonPackageName instanceof SimpleName) {
246 			nonPackageName.accept(this);
247 		}
248 		else {
249 			QualifiedName qualifiedName = (QualifiedName) nonPackageName;
250 			Name qualifier = qualifiedName.getQualifier();
251 
252 			if (hasPackageQualifier(qualifiedName)) {
253 				addTokenToSimpleNamesOfName(qualifier, TokenType.NAMESPACE);
254 			}
255 			else {
256 				nonPackageNameOfImportDeclarationVisitor(qualifier);
257 			}
258 
259 			qualifiedName.getName().accept(this);
260 		}
261 	}

263 	/**
264 	 * Returns whether a qualified name has a package name as its qualifier,
265 	 * even if the qualifier's binding cannot be resolved.
266 	 * This is in order to work around an issue where package names in import statements
267 	 * cannot have their bindings resolved when their corresponding package is not explicitely
268 	 * exported and is part of a module.
269 	 *
270 	 * @param qualifiedName A qualified name.
271 	 * @return {@code true} if the qualified name has a package name as its qualifier,
272 	 * {@code false} otherwise.
273 	 */
274 	private boolean hasPackageQualifier(QualifiedName qualifiedName) {
275 		IBinding qualifierBinding = qualifiedName.getQualifier().resolveBinding();
276 		if (qualifierBinding != null) {
277 			return qualifierBinding instanceof IPackageBinding;
278 		}
279 		else {
280 			IBinding binding = qualifiedName.resolveBinding();
281 			return binding instanceof IPackageBinding || binding instanceof ITypeBinding;
282 		}
283 	}

285 	@Override
286 	public boolean visit(Modifier node) {
287 		addToken(node, TokenType.MODIFIER);
288 		return false;
289 	}

291 	@Override
292 	public boolean visit(SimpleName node) {
293 		IBinding binding = node.resolveBinding();
294 		TokenType tokenType = TokenType.getApplicableType(binding);
295 		if (tokenType != null) {
296 			int modifiers = TokenModifier.getApplicableModifiers(binding);
297 			if (TokenModifier.isGeneric(binding)) {
298 				modifiers |= TokenModifier.GENERIC.bitmask;
299 			}
300 			if (TokenModifier.isDeclaration(node)) {
301 				modifiers |= TokenModifier.DECLARATION.bitmask;
302 			}
303 			addToken(node, tokenType, modifiers);
304 		}
305 
306 		return false;
307 	}

309 	@Override
310 	public boolean visit(ModuleDeclaration node) {
311 		acceptJavdocOfModuleDeclaration(node);
312 		acceptNodeList(node.annotations());
313 		addTokenToSimpleNamesOfName(node.getName(), TokenType.NAMESPACE);
314 		acceptNodeList(node.moduleStatements());
315 		return false;
316 	}

318 	/**
319 	 * Accepts into the Javadoc of a module declaration.
320 	 * This method attempts to work around an issue where {@link ModuleDeclaration#getJavadoc()}
321 	 * always returns {@code null}, even if there is an associated Javadoc comment. This is done by
322 	 * retrieving the Javadoc from {@link CompilationUnit#getCommentList()} instead.
323 	 * This method should be removed once the compiler supports resolving the Javadoc
324 	 * for module declarations.
325 	 *
326 	 * @param moduleDeclaration The module declaration.
327 	 */
328 	private void acceptJavdocOfModuleDeclaration(ModuleDeclaration moduleDeclaration) {
329 		for (Comment comment : this.<Comment>castNodeList(cu.getCommentList())) {
330 			// The start position of the module declaration still includes the Javadoc,
331 			// so we just need to look for a comment with the same start position.
332 			if (comment.getStartPosition() == moduleDeclaration.getStartPosition()) {
333 				comment.accept(this);
334 				break;
335 			}
336 		}
337 	}

378 		acceptNode(node.getExpression());

385 	/**
386 	 * {@link NodeVisitor} for {@link Type} nodes that are type arguments.
387 	 *
388 	 * @param typeArgument
389 	 */
390 	private void typeArgumentVisitor(Type typeArgument) {
391 		visitSimpleNameOfType(typeArgument, (simpleName) -> {
392 			IBinding binding = simpleName.resolveBinding();
393 			TokenType tokenType = TokenType.getApplicableType(binding);
394 			if (tokenType != null) {
395 				int modifiers = TokenModifier.getApplicableModifiers(binding);
396 				if (TokenModifier.isGeneric((ITypeBinding) binding)) {
397 					modifiers |= TokenModifier.GENERIC.bitmask;
398 				}
399 				modifiers |= TokenModifier.TYPE_ARGUMENT.bitmask;
400 				addToken(simpleName, tokenType, modifiers);
401 			}
402 		});
403 	}

405 	@Override
406 	public boolean visit(ClassInstanceCreation node) {
407 		acceptNode(node.getExpression());
408 		visitNodeList(node.typeArguments(), this::typeArgumentVisitor);
409 
410 		visitSimpleNameOfType(node.getType(), (simpleName) -> {
411 			IMethodBinding constructorBinding = node.resolveConstructorBinding();
412 			int modifiers = TokenModifier.getApplicableModifiers(constructorBinding);
413 			if (TokenModifier.isGeneric(constructorBinding) || node.getType().isParameterizedType()) {
414 				modifiers |= TokenModifier.GENERIC.bitmask;
415 			}
416 			addToken(simpleName, TokenType.METHOD, modifiers);
417 		});
418 
419 		acceptNodeList(node.arguments());
420 		acceptNode(node.getAnonymousClassDeclaration());
421 		return false;
422 	}

424 	/**
425 	 * A node visitor may be used by helpers like {@link #visitSimpleNameOfType(Name, NodeVisitor)},
426 	 * and is responsible for the visitation logic of a special kind of node.
427 	 */
428 	@FunctionalInterface
429 	private interface NodeVisitor<T extends ASTNode> {
430 		public void visit(T node);
431 	}

433 	/**
434 	 * Helper method to make an unchecked cast from a raw list of AST nodes
435 	 * to a generic list of a certain type. The caller is responsible for
436 	 * making sure that the cast is safe. An example use case is for
437 	 * {@link CompilationUnit#getCommentList()}, which returns a raw list,
438 	 * but whose Javadoc states that the element type is {@link Comment}.
439 	 *
440 	 * @param <T> The type of {@link ASTNode} expected in the node list.
441 	 * @param nodeList The node list to be cast.
442 	 * @return The cast node list.
443 	 */
444 	@SuppressWarnings("unchecked")
445 	private <T extends ASTNode> List<T> castNodeList(List<?> nodeList) {
446 		return (List<T>) nodeList;
447 	}
448 
449 	/**
450 	 * Helper method which visits the nodes in a raw list of AST nodes
451 	 * using a {@link NodeVisitor}. An unchecked cast is made via
452 	 * {@link #castNodeList(List)}, so the caller is responsible for
453 	 * making sure that the cast is safe.
454 	 *
455 	 * @param <T> The type of {@link ASTNode} expected in the node list.
456 	 * @param nodeList The node list to visit (may be {@code null}).
457 	 * @param visitor A {@link NodeVisitor}.
458 	 */
459 	private <T extends ASTNode> void visitNodeList(List<?> nodeList, NodeVisitor<T> visitor) {
460 		if (nodeList != null) {
461 			for (T node : this.<T>castNodeList(nodeList)) {
462 				visitor.visit(node);
463 			}
464 		}
465 	}
466 
467 	/**
468 	 * Helper method which accepts into the nodes in a raw list of AST nodes.
469 	 * An unchecked cast is made via {@link #castNodeList(List)},
470 	 * so the caller is responsible for making sure that the cast is safe.
471 	 *
472 	 * @param nodeList The node list to accept into (may be {@code null}).
473 	 */
474 	private void acceptNodeList(List<?> nodeList) {
475 		if (nodeList != null) {
476 			for (ASTNode node : this.<ASTNode>castNodeList(nodeList)) {
477 				node.accept(this);
478 			}
479 		}
480 	}

483 	 * Helper method which performs a null-check on and then accepts into an AST node.
484 	 *
485 	 * @param node The AST node to accept into (may be {@code null}).
486 	 */

488 		if (node != null) {
489 			node.accept(this);

493 	/**
494 	 * Helper method to recursively visit all the simple names of a {@link Name} node
495 	 * using the specified {@link NodeVisitor}.
496 	 *
497 	 * @param name A {@link Name} node (may be {@code null}).
498 	 * @param visitor The {@link NodeVisitor} to use.
499 	 */
500 	private void visitSimpleNamesOfName(Name name, NodeVisitor<SimpleName> visitor) {
501 		if (name == null) {
502 			return;
503 		}
504 		if (name instanceof SimpleName) {
505 			visitor.visit((SimpleName) name);
506 		}
507 		else {
508 			QualifiedName qualifiedName = (QualifiedName) name;
509 			visitSimpleNamesOfName(qualifiedName.getQualifier(), visitor);
510 			visitor.visit(qualifiedName.getName());
511 		}
512 	}
513 
514 	/**
515 	 * Helper method to recursively add a token of the specified type to
516 	 * all the simple names of a name.
517 	 *
518 	 * @param name A {@link Name} node (may be {@code null}).
519 	 * @param tokenType The {@link TokenType} to add for all the simple names.
520 	 */
521 	private void addTokenToSimpleNamesOfName(Name name, TokenType tokenType) {
522 		visitSimpleNamesOfName(name, simpleName -> addToken(simpleName, tokenType));
523 	}
524 
525 	/**
526 	 * Helper method to find and visit the simple name of a {@code Type} node
527 	 * using the specified {@link NodeVisitor}.
528 	 *
529 	 * @param type A type node (may be {@code null}).
530 	 * @param visitor The {@link NodeVisitor} to use.
531 	 */
532 	private void visitSimpleNameOfType(Type type, NodeVisitor<SimpleName> visitor) {
533 		if (type == null) {
534 			return;
535 		}
536 		else if (type instanceof SimpleType) {
537 			SimpleType simpleType = (SimpleType) type;
538 			acceptNodeList(simpleType.annotations());
539 
540 			Name simpleTypeName = simpleType.getName();
541 			if (simpleTypeName instanceof SimpleName) {
542 				visitor.visit((SimpleName) simpleTypeName);
543 			}
544 			else {
545 				QualifiedName qualifiedName = (QualifiedName) simpleTypeName;
546 				qualifiedName.getQualifier().accept(this);
547 				visitor.visit(qualifiedName.getName());
548 			}
549 		}
550 		else if (type instanceof QualifiedType) {
551 			QualifiedType qualifiedType = (QualifiedType) type;
552 			qualifiedType.getQualifier().accept(this);
553 			acceptNodeList(qualifiedType.annotations());
554 			visitor.visit(qualifiedType.getName());
555 		}
556 		else if (type instanceof NameQualifiedType) {
557 			NameQualifiedType nameQualifiedType = (NameQualifiedType) type;
558 			nameQualifiedType.getQualifier().accept(this);
559 			acceptNodeList(nameQualifiedType.annotations());
560 			visitor.visit(nameQualifiedType.getName());
561 		}
562 		else if (type instanceof ParameterizedType) {
563 			ParameterizedType parameterizedType = (ParameterizedType) type;
564 			visitSimpleNameOfType(parameterizedType.getType(), visitor);
565 			visitNodeList(parameterizedType.typeArguments(), this::typeArgumentVisitor);
566 		}
567 		else {
568 			type.accept(this);
569 		}
570 	}
